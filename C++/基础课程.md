### 常量(值不允许赋值更改)
```c
#include <stdio.h>
#include <math.h>
int main() {
    const float PI = 3.1415;
    float s;
    s = PI * pow(2,2);
    printf("圆的面积是 %.2f\n", s);
    s = PI * pow(8,2);
    printf("圆的面积是 %.2f\n", s);
    return 0;
}
```
